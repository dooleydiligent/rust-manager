```
#![allow(non_snake_case)]
use dioxus::prelude::*;
use dioxus_free_icons::icons::bs_icons::{BsGearWide, BsList, BsPerson, BsPersonCircle};
use dioxus_free_icons::icons::fa_solid_icons::{FaFile, FaPenToSquare, FaQuestionCircle, FaStar};
use dioxus_free_icons::Icon;

fn main() {
    launch(App);
}

fn App() -> Element {
    let mut is_sidebar_collapsed = use_state(|| false);

    rsx! {
        // We include the CSS generated by Tailwind for styling
        link { rel: "stylesheet", href: "tailwind.css" }

        div { class: "flex min-h-screen bg-gray-100",
            // Collapsible Sidebar
            Sidebar { is_collapsed: is_sidebar_collapsed }

            // Main Content
            div { class: "flex-1 flex flex-col transition-all duration-300 {if *is_sidebar_collapsed { \"ml-20\" } else { \"ml-64\" }}",
                // Top Navigation Bar
                Navbar { is_sidebar_collapsed: is_sidebar_collapsed }
                
                // Page Content Area
                div { class: "p-8",
                    h1 { class: "text-2xl font-bold", "Welcome to the App!" }
                    p { class: "mt-4 text-gray-700", "This is the main content area." }
                }
            }
        }
    }
}

#[component]
fn Navbar(is_sidebar_collapsed: UseState<bool>) -> Element {
    let mut show_profile_menu = use_state(|| false);

    rsx! {
        nav { class: "bg-white shadow-md p-4 flex items-center justify-between",
            div { class: "flex items-center",
                // Hamburger Icon for collapsing the sidebar
                button {
                    class: "p-2 rounded-md hover:bg-gray-200 focus:outline-none",
                    onclick: move |_| is_sidebar_collapsed.set(!is_sidebar_collapsed),
                    Icon { width: 24, height: 24, icon: BsList }
                }
                // Placeholder Menu Items
                DropdownMenu { title: "File" }
                DropdownMenu { title: "Edit" }
                DropdownMenu { title: "Help" }
            }
            // User Profile Button
            div { class: "relative",
                button {
                    class: "p-2 rounded-full flex items-center gap-2 hover:bg-gray-200 focus:outline-none",
                    onclick: move |_| show_profile_menu.set(!show_profile_menu),
                    span { class: "text-gray-700 font-medium", "User" }
                    Icon { width: 24, height: 24, icon: BsPersonCircle }
                }
                // Profile Dropdown
                if *show_profile_menu.get() {
                    ul { class: "absolute right-0 mt-2 w-48 bg-white rounded-md shadow-lg py-1 z-50",
                        li {
                            button { class: "block w-full text-left px-4 py-2 text-gray-800 hover:bg-gray-100", "Profile" }
                        }
                        li {
                            button { class: "block w-full text-left px-4 py-2 text-gray-800 hover:bg-gray-100", "Settings" }
                        }
                    }
                }
            }
        }
    }
}

#[component]
fn DropdownMenu(title: String) -> Element {
    let mut show_menu = use_state(|| false);

    rsx! {
        div { class: "relative ml-4",
            button {
                class: "p-2 rounded-md hover:bg-gray-200 focus:outline-none",
                onclick: move |_| show_menu.set(!show_menu),
                "{title}"
            }
            if *show_menu.get() {
                ul { class: "absolute left-0 mt-2 w-48 bg-white rounded-md shadow-lg py-1 z-50",
                    li { button { class: "block w-full text-left px-4 py-2 text-gray-800 hover:bg-gray-100", "Item 1" } }
                    li { button { class: "block w-full text-left px-4 py-2 text-gray-800 hover:bg-gray-100", "Item 2" } }
                }
            }
        }
    }
}

#[component]
fn Sidebar(is_collapsed: UseState<bool>) -> Element {
    rsx! {
        div {
            class: "flex flex-col bg-gray-800 text-white transition-all duration-300
                    {if *is_collapsed.get() { \"w-20\" } else { \"w-64\" }}",
            div { class: "p-4 flex items-center justify-center",
                h1 {
                    class: "text-xl font-bold transition-all duration-300 {if *is_collapsed.get() { \"opacity-0 w-0\" } else { \"opacity-100 w-auto\" }}",
                    "App Name"
                }
            }
            ul { class: "flex-1 mt-4 space-y-2",
                SidebarItem {
                    icon: BsPerson,
                    text: "Profile".to_string(),
                    is_collapsed: is_collapsed,
                }
                SidebarItem {
                    icon: FaFile,
                    text: "Documents".to_string(),
                    is_collapsed: is_collapsed,
                }
                SidebarItem {
                    icon: FaStar,
                    text: "Favorites".to_string(),
                    is_collapsed: is_collapsed,
                }
                SidebarItem {
                    icon: BsGearWide,
                    text: "Settings".to_string(),
                    is_collapsed: is_collapsed,
                }
            }
        }
    }
}

#[component]
fn SidebarItem(icon: Icon, text: String, is_collapsed: UseState<bool>) -> Element {
    rsx! {
        li {
            a {
                class: "flex items-center p-4 hover:bg-gray-700 transition-colors duration-200",
                Icon { width: 24, height: 24, fill: "white", icon: icon }
                span {
                    class: "ml-4 transition-all duration-300 {if *is_collapsed.get() { \"opacity-0 w-0\" } else { \"opacity-100 w-auto\" }}",
                    "{text}"
                }
            }
        }
    }
}
```
